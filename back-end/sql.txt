CREATE TABLE events (
   event_id serial PRIMARY KEY,
   event_name VARCHAR ( 250 ) UNIQUE NOT NULL,
   place_name VARCHAR ( 150 ),
   link VARCHAR ( 250 ),
   address VARCHAR ( 150 ),
   city VARCHAR ( 150 ),
   start_date TIMESTAMP
);

CREATE TABLE places (
   place_id serial PRIMARY KEY,
   place_name VARCHAR ( 250 ) UNIQUE NOT NULL,
   place_type VARCHAR ( 250 ),
   link VARCHAR ( 250 ),
   address VARCHAR ( 150 ),
   city VARCHAR ( 150 )
);
